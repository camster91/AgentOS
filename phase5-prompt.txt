```kotlin
package com.agentOS.core

import com.agentOS.ai.ChatTurn
import com.agentOS.core.storage.StorageAPI
import com.google.gson.Gson
import com.google.gson.reflect.TypeToken

class ConversationMemory(private val storage: StorageAPI) {
    private val gson = Gson()
    private val maxTurns = 50

    fun getHistory(agentId: String): List<ChatTurn> {
        return try {
            val json = storage.get("memory_$agentId") ?: return emptyList()
            val type = object : TypeToken<List<ChatTurn>>() {}.type
            gson.fromJson(json, type) ?: emptyList()
        } catch (e: Exception) {
            emptyList()
        }
    }

    fun addTurn(agentId: String, turn: ChatTurn) {
        val history = getHistory(agentId).toMutableList()
        history.add(turn)
        if (history.size > maxTurns) {
            history.removeAt(0)
        }
        storage.set("memory_$agentId", gson.toJson(history))
    }
    
    fun clearHistory(agentId: String) {
        storage.delete("memory_$agentId")
    }
}
```
